SELECT Y.UnitID as YS_Unit,U.UnitID as DNA_Unit,COALESCE(U.UnitID,Y.UnitID) as New_Unit_ID,U.UnitName,Y.PROPERTY_NAME,Y.*
FROM "LIV_SANDBOX"."MAPPING"."YS_UNIT" Y
LEFT JOIN (SELECT U.*,P.LIVCORBUPROPERTYIDENTIFIER as BU
FROM "DNA_SDW_PD"."LIV_ODS"."PROPERTYMAPPING_A" P
LEFT JOIN "DNA_SDW_PD"."LIV_ODS"."UNIT_A" U
ON P.PropertyAssetID = U.PropertyAssetID) U
ON Y.BU = U.BU
AND Y.UnitName = U.UnitName

//WHERE U.UnitID <> Y.UnitID

ORDER BY Y.PROPERTY_NAME, U.UnitName
