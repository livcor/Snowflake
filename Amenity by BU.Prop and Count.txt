SELECT
--PMAP.PROPERTYMGMTCOMPANYNAME as Operator, PMAP.PROPERTYNAME as Property_Name, 
DISTINCT AMN.AMENITYNAME as Amenity_Name, COUNT(AMN.AmenityCode) as Amenities
--UMAP.UNITNAME, AMN.AMENITYCODE, AMN.VALIDFROM
FROM "DNA_SDW_DV"."LIV_ODS"."AMENITY" AMN
LEFT JOIN "DNA_SDW_DV"."LIV_ODS"."PROPERTYMAPPING" PMAP
ON AMN.PROPERTYASSETID = PMAP.PROPERTYASSETID 
LEFT JOIN "DNA_SDW_DV"."LIV_ODS"."UNIT" UMAP
ON AMN.UNITID = UMAP.UNITID
LEFT JOIN DNA_SDW_DV.LIV_ODS.AmenityValue as AV
ON AMN.AmenityID = AV.AmenityID

WHERE AMN.ACTIVEIND = 1 
AND AMN.DL_ISCURRENT = 1
AND PMAP.DL_ISCURRENT = 1
AND PMAP.ACTIVEIND = 1
AND UMAP.ACTIVEIND = 1
AND UMAP.DL_ISCURRENT = 1 
GROUP BY --PMAP.PROPERTYMGMTCOMPANYNAME, PMAP.PROPERTYNAME, 
AMN.AMENITYNAME
ORDER BY --PMAP.PROPERTYMGMTCOMPANYNAME, PMAP.PROPERTYNAME, 
Amenities desc


SELECT
PMAP.PROPERTYMGMTCOMPANYNAME as Operator, PMAP.PROPERTYNAME as Property_Name, PMAP.LIVCORBUPROPERTYIDENTIFIER as BU, AMN.AMENITYNAME as Amenity_Name, COUNT(AMN.AmenityCode) as Amenities

--UMAP.UNITNAME, AMN.AMENITYCODE, AMN.VALIDFROM
FROM "DNA_SDW_DV"."LIV_ODS"."AMENITY" AMN
LEFT JOIN "DNA_SDW_DV"."LIV_ODS"."PROPERTYMAPPING" PMAP
ON AMN.PROPERTYASSETID = PMAP.PROPERTYASSETID 
LEFT JOIN "DNA_SDW_DV"."LIV_ODS"."UNIT" UMAP
ON AMN.UNITID = UMAP.UNITID
WHERE AMN.ACTIVEIND = 1 
AND AMN.DL_ISCURRENT = 1
AND PMAP.ACTIVEIND = 1
AND PMAP.DL_ISCURRENT = 1
AND UMAP.ACTIVEIND = 1
AND UMAP.DL_ISCURRENT = 1 
GROUP BY PMAP.PROPERTYMGMTCOMPANYNAME, PMAP.PROPERTYNAME,PMAP.LIVCORBUPROPERTYIDENTIFIER, AMN.AMENITYNAME
ORDER BY PMAP.PROPERTYMGMTCOMPANYNAME, PMAP.PROPERTYNAME, PMAP.LIVCORBUPROPERTYIDENTIFIER, AMN.AMENITYNAME
